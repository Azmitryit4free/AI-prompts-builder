  if(neg) document.getElementById('neg-out').value=neg.replace(/^No /i,'').toLowerCase()+', deformed, ugly, bad anatomy, watermark, signature';

  else document.getElementById('neg-out').value='';

 

  updateCharCount();

  updateNegCount();

  updateSuffix();

  updateProgress();

}

 

function updateCharCount(){document.getElementById('char-count').textContent=document.getElementById('prompt-out').value.length+' characters';}

function updateNegCount(){document.getElementById('neg-count').textContent=document.getElementById('neg-out').value.length+' characters';}

 

function copyAll(){

  const p=document.getElementById('prompt-out').value;

  const suf=document.getElementById('suffix-out').textContent;

  const safeSuf=suf.startsWith('Paste')||suf.startsWith('Select')?'':'\n\n'+suf;

  if(!p){alert('Nothing to copy yet! Select some options first.');return;}

  navigator.clipboard.writeText(p+safeSuf).then(()=>{

    const b=event.target;const orig=b.textContent;b.textContent='✅ Copied!';setTimeout(()=>b.textContent=orig,2000);

  }).catch(()=>{});

}

 

function copyAndOpen(m){

  const p=document.getElementById('prompt-out').value;

  if(!p){alert('Build a prompt first!');return;}

  const suf=modelSuffixes[m].replace(/\{ratio\}/g,getRatio());

  navigator.clipboard.writeText(p+' '+suf).then(()=>{

    const b=event.target;const orig=b.textContent;b.textContent='✅ Copied for Midjourney!';setTimeout(()=>b.textContent=orig,2000);

  }).catch(()=>{});

}

 

function savePrompt(){

  const p=document.getElementById('prompt-out').value.trim();

  if(!p){alert('Nothing to save yet!');return;}

  saved.unshift({text:p,model:currentModel,mode,time:new Date().toLocaleTimeString()});

  if(saved.length>10) saved.pop();

  renderSaved();

}

 

function renderSaved(){

  const list=document.getElementById('saved-list');

  if(!saved.length){list.innerHTML='<div class="empty-saved">No saved prompts yet — build one above and click Save.</div>';return;}

  list.innerHTML=saved.map((s,i)=>`

    <div class="saved-item" onclick="loadSaved(${i})">

      <div style="flex:1;min-width:0">

        <div class="saved-meta">${s.mode} · ${s.model} · ${s.time}</div>

        <div class="saved-text">${s.text}</div>

      </div>

      <button class="saved-del" onclick="event.stopPropagation();deleteSaved(${i})" title="Delete">✕</button>

    </div>`).join('');

}

 

function loadSaved(i){document.getElementById('prompt-out').value=saved[i].text;updateCharCount();}

function deleteSaved(i){saved.splice(i,1);renderSaved();}

function clearSaved(){if(confirm('Clear all saved prompts?')){saved=[];renderSaved();}}

 

function resetAll(){

  if(!confirm('Reset all selections?')) return;

  document.querySelectorAll('select').forEach(s=>{s.selectedIndex=0;s.classList.remove('filled');});

  document.getElementById('prompt-out').value='';

  document.getElementById('neg-out').value='';

  document.getElementById('suffix-out').textContent='Select a model above to see the recommended suffix.';

  updateCharCount();updateNegCount();updateProgress();

}
